      #定义机械压缩函数
      def yasuo(st):
          for i in range(1,int(len(st)/2)+1):
              for j in range(len(st)):
                  if st[j:j+i] == st[j+i:j+2*i]:
                      k = j + i
                      while st[k:k+i] == st[k+i:k+2*i] and k<len(st):   
                          k = k + i
                      st = st[:j] + st[k:]    
          return st
