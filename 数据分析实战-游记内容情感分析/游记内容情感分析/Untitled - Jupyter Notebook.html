      import string
      def str_count(str):
          count_en = count_dg = count_sp = count_zh = count_pu = 0
          for s in str:
              if s in string.ascii_letters:
                  count_en += 1
              elif s.isdigit():
                  count_dg += 1
              elif s.isspace():
                  count_sp += 1
              elif s.isalpha():
                  count_zh += 1
              else:
                  count_pu += 1
          return count_zh
